1  Listar os empregados (nomes) que tem salário maior que seu chefe (usar o join)
--  empregado | 
-- -----------+
--  Maria     |
--  Claudia   |
--  Ana       |
--  Luiz      |

select func.nome from empregados func join empregados sup on func.supervisor_id=sup.emp_id where func.salario > sup.salario;

2 Listar o maior salario de cada departamento (usa o group by )
--  dep_id |  max  
-- --------+-------
--       1 | 10000
--       2 |  8000
--       3 |  6000
--       4 | 12200

select dep_id, max(salario) from empregados group by dep_id;
